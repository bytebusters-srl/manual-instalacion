% 01_packages.tex
% Paquetes esenciales para el cumplimiento de ISO/IEC/IEEE 15289:2024 e IEEE 830
% (Manual de Instalación — Plataforma EthosHub)

% Codificación e idioma
\usepackage[utf8]{inputenc}
\usepackage[T1]{fontenc}
\usepackage[spanish,es-tabla,es-noindentfirst]{babel} % Soporte en español

% Familias Tipográficas (Norma de Diseño)
\usepackage{mathptmx}  % Cuerpo: Serif (Times New Roman)
\usepackage{helvet}    % Títulos: Sans-serif (Helvetica)
\usepackage{courier}   % Código: Monospace (Courier New / Lucida)
\usepackage[protrusion=true,expansion=true]{microtype} % Microtipografía: elimina overfulls leves

% Geometría y Layout
\usepackage{geometry}  % Control preciso de los márgenes de página
\usepackage{setspace}  % Configuración de interlineado
\usepackage{fancyhdr}  % Personalización de encabezados y pies de página
\usepackage{lastpage}  % Referencia al total de páginas
\usepackage{titlesec}  % Formateo de títulos y subtítulos

% Gráficos, Colores y Símbolos
\usepackage{graphicx}      % Inserción de imágenes y logos
\usepackage{xcolor}        % Definición de paleta de colores corporativa
\usepackage{fontawesome5}  % Iconografía para avisos técnicos
\usepackage{amssymb}       % Símbolos para checklists

% Tablas Técnicas (Matrices de pre-requisitos, variables y compatibilidad)
\usepackage{tabularx}      % Tablas autoajustables
\usepackage{colortbl}      % Color de fondo para celdas/filas de tablas
\usepackage{booktabs}      % Líneas profesionales para tablas
\usepackage{multirow}      % Celdas combinadas en tablas
\usepackage{longtable}     % Tablas multipágina (matriz de variables, errores)
\usepackage{array}         % Columnas con formato avanzado

% Cajas de Diseño y Listas (avisos, alertas, checklists)
\usepackage[most]{tcolorbox} % Cajas visuales (AvisoNota, AlertaSeguridad, ...)
\usepackage{enumitem}        % Listas personalizadas (checklist, noitemsep)

% Diagramas Vectoriales Nativos (topologías de red, flujos de despliegue, mockups)
\usepackage{tikz}
\usetikzlibrary{
    positioning,      % Posicionamiento relativo de nodos
    shapes.geometric, % Rombos (decisiones), cilindros (BBDD)
    shapes.symbols,
    shapes.multipart,
    arrows.meta,      % Puntas de flecha modernas
    fit,              % Recuadros que envuelven grupos de nodos
    calc,             % Cálculo de coordenadas
    backgrounds,      % Capas de fondo para fronteras de contenedores
    decorations.pathreplacing,
    chains
}

% Bloques de Código Fuente / Terminal (Bash, YAML, ENV, JSON, SQL)
\usepackage{listings}

% Rutas y URLs largas con quiebre de línea automático (evita overfull en
% direcciones, paquetes y rutas de archivo del tipo /api/v1/... o chrome://flags)
\usepackage{seqsplit}      % Quiebre carácter a carácter de cadenas largas
\usepackage{ragged2e}      % Justificado flexible para celdas estrechas

% Navegación y Metadatos PDF
\usepackage{hyperref}  % Generación de enlaces internos
\usepackage{xurl}      % Quiebre de URLs en cualquier carácter (tras hyperref)
