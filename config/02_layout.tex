% 1. Configuración de Márgenes Exactos
\geometry{
    a4paper,
    inner=2.5cm,       % Margen interno para archivo físico
    outer=2.5cm,       % Margen externo
    top=2.5cm,
    bottom=2.5cm,
    includehead,
    includefoot,
    headheight=28pt,
    headsep=0.4cm,
    footskip=1.0cm
}
\onehalfspacing      % Interlineado 1.5 para legibilidad en lectura larga

% Holgura de último recurso: permite aflojar líneas con cajas inquebrantables
% (\comando, \ruta, URLs) en vez de invadir el margen con desbordes (overfull).
\setlength{\emergencystretch}{4em}

% Evitar el estiramiento vertical (\flushbottom por defecto en twoside) que, ante
% un float o sección que no cabe, reparte hasta varios cm de espacio en blanco
% entre párrafos. Con \raggedbottom el contenido se asienta arriba y el sobrante
% queda al pie, sin huecos internos antiestéticos.
\raggedbottom

% 2. Variables de trazabilidad documental (ISO 15289). Se redefinen desde main.tex.
\newcommand{\currentDocID}{MI-INST-000}
\newcommand{\currentDocEstado}{Borrador}
\newcommand{\currentAmbiente}{Producción}
\newcommand{\setDocID}[1]{\renewcommand{\currentDocID}{#1}}
\newcommand{\setDocEstado}[1]{\renewcommand{\currentDocEstado}{#1}}
\newcommand{\setAmbiente}[1]{\renewcommand{\currentAmbiente}{#1}}

% 3. Jerarquía de Títulos (estilo "libro": portada de capítulo en plana propia)
\titleformat{\chapter}[display]
  {\sffamily\bfseries\color{colorPrimario}}
  {\filright\fontsize{13}{15}\selectfont\color{grisISO}\MakeUppercase{Capítulo \thechapter}}
  {0.35cm}
  {\fontsize{25}{29}\selectfont\filright}
  []
\titlespacing*{\chapter}{0pt}{55pt}{24pt}

% \resumenCapitulo{texto}: sinopsis breve bajo el título del capítulo, en la
% misma plana. Cierra con \clearpage para que el contenido empiece en la
% página siguiente (estilo libro).
\newcommand{\resumenCapitulo}[1]{%
  \vspace{0.5cm}%
  {\color{colorAcento}\rule{2.5cm}{2pt}}\par
  \vspace{0.45cm}%
  {\normalfont\sffamily\fontsize{11}{15}\selectfont\color{grisISO!95!black}%
   \itshape #1\par}%
  \clearpage}

% Subtítulos: Helvetica
\titleformat{\section}[hang]
  {\sffamily\bfseries\fontsize{14}{16.8}\selectfont\color{colorPrimario}}
  {\thesection.}
  {1em}{}
\titleformat{\subsection}[hang]
  {\sffamily\bfseries\fontsize{12}{14.4}\selectfont\color{colorPrimario!85!black}}
  {\thesubsection.}
  {0.7em}{}

% 4. Identidad Corporativa: Encabezado y Pie (Trazabilidad de Instalación)

% Redefinimos marcas para capturar el nombre limpio del capítulo
\renewcommand{\chaptermark}[1]{\markboth{#1}{}}

\fancyfoot[R]{\sffamily\footnotesize\textcolor{grisISO}{Página \thepage\ de~\pageref{LastPage}}}
